Interprétation des types

double **ptr ;	/* pointeur vers pointeur double */

int *t[10] ;	/* tableau de pointeurs vers int */
int (*t)[10] ;	/* pointeur vers un tableau d'entiers */

int *f() ;	/* fonction renvoyant un pointeur vers int */
int (*f)() ;	/* pointeur vers une fonction renvoyant un int */

/* fonction renvoyant un pointeur vers pointeur de int */
int **f() ;

/* pointeur vers une fonction renvoyant un pointeur de int */
int *(*f)() ;

/* pointeur de pointeur vers une fonction renvoyant un int */
int (**f)() ;

/* tableau de pointeurs vers des fonctions */
void (*t[100])(int) ;

/* fonction renvoyant un pointeur vers une fonction renvoyant un double */
double (*g(int))(char) ;
